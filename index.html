async function obtenerDatos() { const input = document.getElementById('equipoA'); const resDiv = document.getElementById('resultado'); const nombreBusqueda = input ? input.value.toLowerCase() : ""; if (!nombreBusqueda) { resDiv.innerText = "Por favor, escribe el nombre del equipo."; return; } resDiv.innerText = "Consultando..."; const apiKey = "TU_CLAVE_REAL_AQUI"; const targetUrl = `https://api.sportmonks.com/v3/football/teams?api_token=${apiKey}`; const proxyUrl = "https://corsproxy.io/?"; try { const respuesta = await fetch(proxyUrl + encodeURIComponent(targetUrl)); const json = await respuesta.json(); // REVISIÓN: ¿Qué recibimos realmente? console.log(json); if (json && json.data) { const equipo = json.data.find(e => e.name.toLowerCase().includes(nombreBusqueda)); resDiv.innerText = equipo ? `Equipo: ${equipo.name}` : "No encontrado"; } else { resDiv.innerText = "Error: La API no envió equipos. Revisa tu API Key."; } } catch (e) { resDiv.innerText = "Error crítico: " + e.message; } }
