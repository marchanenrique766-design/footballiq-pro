async function obtenerDatos() { const A = document.getElementById('equipoA').value; const resDiv = document.getElementById('resultado'); if (!A) { resDiv.innerText = "Escribe el nombre del equipo."; return; } resDiv.innerText = "Analizando datos..."; const apiKey = "TU_CLAVE_REAL_AQUI"; // Usamos un endpoint más completo de Sportmonks (asegúrate que tu plan lo permita) const targetUrl = `https://api.sportmonks.com/v3/football/teams/search/${A}?api_token=${apiKey}&include=statistics`; const proxyUrl = "https://corsproxy.io/?"; try { const respuesta = await fetch(proxyUrl + encodeURIComponent(targetUrl)); const datos = await respuesta.json(); if (datos.data && datos.data.length > 0) { const equipo = datos.data[0]; // Aquí personalizamos qué queremos mostrar resDiv.innerHTML = `${equipo.name}
ID: ${equipo.id}
País: ${equipo.country_id}`; } else { resDiv.innerText = "No se encontraron datos para: " + A; } } catch (error) { resDiv.innerText = "Error: " + error.message; } }
