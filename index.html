async function obtenerDatos() { const nombreBusqueda = document.getElementById('equipoA')?.value?.toLowerCase(); const resDiv = document.getElementById('resultado'); if (!nombreBusqueda) { resDiv.innerText = "Escribe un equipo"; return; } resDiv.innerText = "Consultando..."; const apiKey = "TU_CLAVE_REAL_AQUI"; const targetUrl = `https://api.sportmonks.com/v3/football/teams?api_token=${apiKey}`; const proxyUrl = "https://corsproxy.io/?"; try { const respuesta = await fetch(proxyUrl + encodeURIComponent(targetUrl)); const json = await respuesta.json(); // Validación segura if (json && json.data && Array.isArray(json.data)) { const equipo = json.data.find(e => e.name.toLowerCase().includes(nombreBusqueda)); resDiv.innerText = equipo ? `Equipo: ${equipo.name} (ID: ${equipo.id})` : "No encontrado"; } else { resDiv.innerText = "Error en la estructura de datos."; } } catch (e) { resDiv.innerText = "Error: " + e.message; } }
