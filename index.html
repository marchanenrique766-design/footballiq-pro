async function obtenerDatos() { const A = document.getElementById('equipoA').value; const resDiv = document.getElementById('resultado'); if (!A) { resDiv.innerText = "Por favor, escribe el nombre del equipo."; return; } resDiv.innerText = "Consultando..."; // CAMBIA ESTO POR TU CLAVE REAL const apiKey = "TU_CLAVE_REAL_AQUI"; const targetUrl = `https://api.sportmonks.com/v3/football/teams/search/${A}?api_token=${apiKey}`; const proxyUrl = "https://corsproxy.io/?"; try { const respuesta = await fetch(proxyUrl + encodeURIComponent(targetUrl)); // Si la respuesta no es exitosa, esto nos dirá qué pasó exactamente if (!respuesta.ok) { resDiv.innerText = "Error del servidor: " + respuesta.status; return; } const datos = await respuesta.json(); // Si la estructura del JSON es diferente, aquí lo veremos escrito resDiv.innerText = "Respuesta recibida: " + JSON.stringify(datos); } catch (error) { resDiv.innerText = "Error de conexión: " + error.message; } }
